Pinout:

esp32:

    NSS - 5
    RESET - 15
    DIO0 - любой // не подключен DIO0 pin is optional, it is only needed for receive callback mode
    SCK - 18
    MISO - 19
    MOSI - 23
  
  
atmega:

    NSS - 4
    RESET - 7
    DIO0 - любой // не подключен DIO0 pin is optional, it is only needed for receive callback mode
    SCK - 13
    MISO - 12
    MOSI - 11
